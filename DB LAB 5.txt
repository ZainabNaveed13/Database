SELECT DNAAME, ENAME, LOC
FROM EMP, DEPT
WHERE EMP.DEPTNO = DEPT.DEPTNO
ANE EMP.COMM IS NOT NULL;

2.
SELECT MOD(SAL,300) AS "REMAINDER"
FROM emp
WHERE EMP.JOB = "Manager";

3.
SELECT HIREDATE, COMM, DNAME, LOC, SAL, GRADE
FROM EMP, DEPT, SALGRADE
WHERE EMP.SAL BETWEEN HISAL AND LOSAL
AND DEPT.DEPTNO = EMP.DEPTNO;

4.
SELECT COUNT(*), ROUND(MAX(sal),0) AS "Highest" , ROUND(MIN(sal),0) AS "Lowest", ROUND(SUM(sal),0) AS "SUM", ROUND(AVG(sal),0) AS "AVERAGE"
FROM EMP;

5.
SELECT SUM(SAL), AVG(SAL)
FROM EMP 
GROUP BY DEPTNO
HAVING DEPTNO = 10;

6.
SELECT d.DEPTNO, d.DNAME, d.LOC
FROM dept d, emp e
WHERE d.DEPTNO = e.DEPTNO
GROUP BY DEPTNO
HAVING AVG(SAL) > ALL( SELECT (AVG(SAL) 
		       FROM emp
		       GROUP BY DEPTNO); 
7.
SELECT M.ENAME AS "Managers", M.ENAME AS "Direct Reports"
FROM EMP E, EMP M
WHERE E.MGR = M.EMPNO;


8.
SELECT ENAME, JOB, SAL
FROM EMP, DEPT 
WHERE  EMP.DEPTNO = DEPT.DEPTNO
AND EMP.SAL IN SELECT(MAX(SAL) FROM EMP GROUP BY DEPTNO);

9.
SELECT E.ENAME, E.JOB, E.SAL
FROM EMP E, EMP M
WHERE E.MGR = E.EMPNO
AND E.SAL > M.SAL;

10.
SELECT E.ENAME, E.HIREDATE, M.ENAME
FROM EMP E, EMP M
WHERE E.MGR = M.EMPNO
AND E.HIREDATE < M.HIREDATE;
