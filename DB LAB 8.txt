1.
SELECT E.ENAME, E.DEPTNO 
FROM EMP E
WHERE E.DEPTNO = (SELECT M.DEPTNO
	         FROM EMP M
		 WHERE M.ENAME = "MARTIN");
2.
SELECT M.ENAME
FROM EMP E, EMP M
WHERE E.MGR = M.EMPNO
AND M.SAL = SELECT MAX(Y.SAL)
	    FROM EMP X, EMP Y
	    WHERE X.MGR = Y.EMPNO;

3.
SELECT E.EMPID, E.FIRSTNAME, E.LASTNAME, E.SAL, D.DNAME, L.CITY, L.STATE
FROM EMP E, DEPT D, LOC L
WHERE E.DEPTNO = D.DEPTNO
AND D.LOC_ID = L.LOC_ID
AND E.SAL = (SELECT MAX(M.SAL)
	     FROM EMP M
	     WHERE E.DEPTNO = M.DEPTNO
	     GROUP BY DEPTNO);


4.
SELECT D.DNAME, D.DEPTNO, D.LOC
FROM EMP E, DEPT D
WHERE E.DEPTNO = D.DEPTNO
AND E.MAX(SAL) > 7000
AND COUNT(E.JOB) >= 1
GROUP BY DEPTNO;

5.
SELECT D.DNAME, AVG(E.SAL), COUNT(ENAME)
FROM EMP E, DEPT D
WHERE D.DEPTNO = E.DEPTNO
AND E.COMM IS NOT NULL
GROUP BY DEPTNO;

6.
SELECT *
FROM EMP E, SALGRADE S
WHERE E.SAL BETWEEN (LOSAL AND HISAL)
AND E.JOB IN ('MANAGER','ANALYST')
AND S.GRADE IN (4,5);

7.
SELECT * 
FROM EMP E, EMP M
WHERE E.JOB = M.JOB
AND E.DEPTNO = 20
AND M.DEPTNO = 10;

8.
SELECT E.ENAME
FROM EMP E, DEPT D
WHERE E.DEPTNO = D.DEPTNO
AND D.DNAME = 'SALES'
AND E.SAL = SELECT (MAX(M.SAL)
	    FROM EMP M, DEPT Y
	    WHERE M.DEPTNO = Y.DEPTNO
	    AND M.DNAME = 'SALES');

10.
SELECT *
FROM EMP E
WHERE E.SAL = (SELECT MIN(SAL)
	      FROM EMP M
              WHERE E.JOB = M.JOB
	      GROUP BY M.JOB)

ORDER BY  E.SAL ASC;
